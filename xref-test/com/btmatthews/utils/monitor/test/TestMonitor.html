49      /**
50       * Mock the server test fixture.

52      @Mock
53      private Server server;

56       * Mock the logger test fixture.

58      @Mock
59      private Logger logger;

61      /**
62       * Mock the observer test fixture.
63       */
64      @Mock
65      private MonitorObserver observer;

67      /**
68       * Verify that a monitor can be started and stopped successfully.
69       *
70       * @throws Exception If the test case fails.
71       */
72      @Test
73      void testMonitor() throws Exception {
74          when(server.isStarted(any(Logger.class))).thenReturn(true);
75          when(server.isStopped(any(Logger.class))).thenReturn(true);
76          final Monitor monitor = new Monitor("test", 10000);
77          final Thread monitorThread = monitor.runMonitorDaemon(server, logger, observer);
78          runWithDelay(() -> monitor.sendCommand("stop", logger));
79          monitorThread.join(15000L);
80          verify(server).start(logger);
81          verify(server, times(2)).isStarted(logger);
82          verify(logger).logInfo("Waiting for command from client");
83          verify(logger).logInfo("Sending command \"stop\" to monitor");
84          verify(logger).logInfo("Receiving command from client");
85          verify(server).stop(logger);
86          verify(server).isStopped(logger);
87          verify(observer).started(server, logger);
88          verify(observer).stopped(server, logger);
89          verifyNoMoreInteractions(logger, server, observer);
90      }
91  
92      /**
93       * Verify that a server can be configured via the monitor.
94       *
95       * @throws Exception If the test case fails.
96       */
97      @Test
98      void testMonitorConfigure() throws Exception {
99          when(server.isStarted(any(Logger.class))).thenReturn(true);
100         when(server.isStopped(any(Logger.class))).thenReturn(true);
101         final Monitor monitor = new Monitor("test", 10000);
102         final Thread monitorThread = monitor.runMonitorDaemon(server, logger, observer);
103         runWithDelay(() -> {
104             Monitor.sendCommand("test", 10000, "configure debug=off", logger);
105             Monitor.sendCommand("test", 10000, "stop", logger);
106         });
107         monitorThread.join(15000L);
108         verify(server).start(logger);
109         verify(server, times(2)).isStarted(logger);
110         verify(logger, times(2)).logInfo("Waiting for command from client");
111         verify(logger).logInfo("Sending command \"configure debug=off\" to monitor");
112         verify(logger).logInfo("Sending command \"stop\" to monitor");
113         verify(logger, times(2)).logInfo("Receiving command from client");
114         verify(server).configure("debug","off", logger);
115         verify(server).stop(logger);
116         verify(server).isStopped(logger);
117         verify(observer).started(server, logger);
118         verify(observer).stopped(server, logger);
119         verifyNoMoreInteractions(logger, server, observer);
120     }

122     /**
123      * Verify that the server ignores commands with an invalid key.
124      *
125      * @throws Exception If the test case fails.
126      */
127     @Test
128     void testInvalidMonitorKey() throws Exception {
129         when(server.isStarted(any(Logger.class))).thenReturn(true);
130         when(server.isStopped(any(Logger.class))).thenReturn(true);
131         final Monitor monitor = new Monitor("test", 10000);
132         final Thread monitorThread = monitor.runMonitorDaemon(server, logger, observer);
133         runWithDelay(() -> {
134             Monitor.sendCommand("TEST", 10000, "stop", logger);
135             Monitor.sendCommand("test", 10000, "stop", logger);
136         });
137         monitorThread.join(15000L);
138         verify(server).start(logger);
139         verify(server, times(2)).isStarted(logger);
140         verify(logger, times(2)).logInfo("Waiting for command from client");
141         verify(logger, times(2)).logInfo("Sending command \"stop\" to monitor");
142         verify(logger, times(2)).logInfo("Receiving command from client");
143         verify(logger).logError("Invalid monitor key");
144         verify(server).stop(logger);
145         verify(server).isStopped(logger);
146         verify(observer).started(server, logger);
147         verify(observer).stopped(server, logger);
148         verifyNoMoreInteractions(logger, server, observer);
149     }

151     /**
152      * Verify that the monitor will log an error if it could nt open a TCP port.
153      *
154      * @throws IOException If the test case fails.
155      */
156     @Test
157     void testRunMonitorWithIOException() throws IOException {
158         try (final ServerSocket serverSocket1 = new ServerSocket();
159              final ServerSocket serverSocket2 = new ServerSocket()) {
160             serverSocket1.setReuseAddress(true);
161             serverSocket2.setReuseAddress(true);
162             serverSocket1.bind(new InetSocketAddress(InetAddress.getLocalHost(), 10000), 1);
163             serverSocket2.bind(new InetSocketAddress("localhost", 10000), 1);
164             final Monitor monitor = new Monitor("test", 10000);
165             monitor.runMonitor(server, logger, observer);
166             verify(logger).logError(eq("Error starting or stopping the monitor"), any(IOException.class));
167             verifyNoMoreInteractions(logger, server, observer);
168         }
169     }
