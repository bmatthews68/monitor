27  /**
28   * Unit tests for the Mojo that implements the stop goal.
29   *
30   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
31   * @since 1.1.0

33  @ExtendWith(MockitoExtension.class)
34  class TestStopMojo extends AbstractMonitorTest {
35  
36      /**
37       * Spy test fixture.
38       */
39      @Spy
40      private final AbstractStopMojo mojo = new AbstractStopMojo() {
41      };
42      /**
43       * Spy test fixture.
44       */
45      @Spy
46      private final AbstractStopMojo wrongMojo = new AbstractStopMojo() {
47      };
48      /**
49       * Mock for the LDAP server.
50       */
51      @Mock
52      private Server server;
53      /**
54       * Mock for the logger.
55       */
56      @Mock
57      private Logger logger;
58      /**
59       * Mock observer fixture.
60       */
61      @Mock
62      private MonitorObserver observer;
63      /**
64       * Mock logger for used by Maven.
65       */
66      @Mock
67      private Log log;
68  
69      /**
70       * Prepare for test case execution by initialising the mocks.
71       *
72       * @throws Exception If there was a problem preparing the test cases.
73       */
74      @BeforeEach
75      void setUp() throws Exception {
76          ReflectionUtils.setVariableValueInObject(mojo, "monitorPort", 10000);
77          ReflectionUtils.setVariableValueInObject(mojo, "monitorKey", "dummy");
78          ReflectionUtils.setVariableValueInObject(wrongMojo, "monitorPort", 10000);
79          ReflectionUtils.setVariableValueInObject(wrongMojo, "monitorKey", "jester");
80      }
81  
82      /**
83       * Verify that stop logs an error if there is no monitor running.
84       */
85      @Test
86      void testStopWithNoServer() {
87          when(mojo.getLog()).thenReturn(log);
88          mojo.execute();
89          verify(log).info("Sending command \"stop\" to monitor");
90          verify(log).error(eq("Error sending command to monitor"), any(IOException.class));
91          verifyNoMoreInteractions(server, logger, log, observer);
92      }
93  
94      /**
95       * Start a mock server and verify that the {@link com.btmatthews.utils.monitor.mojo.AbstractStopMojo} signals it to shutdown.
96       *
97       * @throws Exception If the test case failed.
98       */
99      @Test
100     void testStopWithRunningServer() throws Exception {
101         when(mojo.getLog()).thenReturn(log);
102         when(server.isStarted(any(Logger.class))).thenReturn(true);
103         when(server.isStopped(any(Logger.class))).thenReturn(true);
104         final Monitor monitor = new Monitor("dummy", 10000);
105         final Thread monitorThread = monitor.runMonitorDaemon(server, logger, observer);
106         runWithDelay(mojo::execute);
107         monitorThread.join(15000L);
108         verify(server).start(logger);
109         verify(server, times(2)).isStarted(logger);
110         verify(logger).logInfo("Waiting for command from client");
111         verify(log).info("Sending command \"stop\" to monitor");
112         verify(logger).logInfo("Receiving command from client");
113         verify(server).stop(logger);
114         verify(server).isStopped(logger);
115         verify(observer).started(server, logger);
116         verify(observer).stopped(server, logger);
117         verifyNoMoreInteractions(server, logger, log, observer);
118     }
119 
120     /**
121      * Verify that stop logs an error if the wrong client is used.
122      *
123      * @throws Exception If the unit test failed.
124      */
125     @Test
126     void testStopWithWrongMojo() throws Exception {
127         when(mojo.getLog()).thenReturn(log);
128         when(wrongMojo.getLog()).thenReturn(log);
129         when(server.isStarted(any(Logger.class))).thenReturn(true);
130         when(server.isStopped(any(Logger.class))).thenReturn(true);
131         final Monitor monitor = new Monitor("dummy", 10000);
132         final Thread monitorThread = monitor.runMonitorDaemon(server, logger, observer);
133         runWithDelay(() -> {
134             wrongMojo.execute();
135             mojo.execute();
136         });
137         monitorThread.join(15000L);
138         verify(server).start(logger);
139         verify(server, times(2)).isStarted(logger);
140         verify(logger, times(2)).logInfo("Waiting for command from client");
141         verify(log, times(2)).info("Sending command \"stop\" to monitor");
142         verify(logger, times(2)).logInfo("Receiving command from client");
143         verify(logger).logError("Invalid monitor key");
144         verify(server).stop(logger);
145         verify(server).isStopped(logger);
146         verify(observer).started(server, logger);
147         verify(observer).stopped(server, logger);
148         verifyNoMoreInteractions(server, logger, log, observer);
149     }
